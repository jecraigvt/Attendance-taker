```javascript
function getPeriodStartTime(period, schedule) {
  const sched = bellSchedules[schedule] || bellSchedules.Default;
  const timeStr = sched[period];
  if (!timeStr) return null;

  const [h, m] = timeStr.split(':').map(x => parseInt(x, 10));
  const d = new Date();
  d.setHours(h, m, 0, 0);
  return d;
}
```

```javascript
// ===== Attendance helpers =====

function statusForNow(pendingEntry = null) {
  // Get the current period from the UI
  const p = periodSelect.value;
  if (!p) return 'On Time';  // No period selected, default to on time

  // Get the period start time from bell schedule
  const periodStart = getPeriodStartTime(p, scheduleName);
  if (!periodStart) {
    // Period not in schedule (edge case), fall back to on-time
    console.warn(`[statusForNow] Period ${p} not found in ${scheduleName} schedule`);
    return 'On Time';
  }

  // Calculate the tardy threshold
  const tardyThreshold = new Date(periodStart.getTime() + TARDY_GRACE_MINUTES * 60 * 1000);

  // Compare current time to threshold
  const now = new Date();

  if (now <= tardyThreshold) {
    return 'On Time';
  }

  return 'Late';
}
```
