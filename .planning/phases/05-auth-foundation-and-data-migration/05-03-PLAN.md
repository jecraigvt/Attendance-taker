```javascript
async function initFirebase() {
  if (!FB_CONFIG) {
    rosterStatus.textContent = 'Local Mode: data saved in this browser.';
    rosterStatus.classList.add('text-green-600');
    return;
  }
  try {
    const app = initializeApp(FB_CONFIG);
    try {
      db = initializeFirestore(app, { localCache: persistentLocalCache() });
    } catch (persistErr) {
      db = getFirestore(app);
    }
    auth = getAuth(app);

    // Check for existing kiosk binding
    const storedUid = localStorage.getItem('kioskTeacherUid');
    const storedName = localStorage.getItem('kioskTeacherName');
    const storedPin = localStorage.getItem('kioskTeacherPin');

    if (storedUid && storedPin) {
      // Kiosk is already bound — try to restore session
      try {
        // Re-authenticate with stored custom token if available
        const storedToken = localStorage.getItem('kioskAuthToken');
        if (storedToken) {
          await signInWithCustomToken(auth, storedToken);
        }
        teacherUid = storedUid;
        teacherName = storedName || 'Teacher';
        teacherPin = storedPin;
        showKioskMode();
      } catch (authErr) {
        console.warn('[initFirebase] Stored token expired, showing login');
        showLoginScreen();
      }
    } else {
      showLoginScreen();
    }
  } catch (err) {
    console.error(err);
    rosterStatus.textContent = 'Cloud error — Local Mode enabled.';
    db = null; auth = null;
  }
}
```

**G. Add login handler:**
```javascript
async function handleTeacherLogin() {
  const username = document.getElementById('aeries-username').value.trim();
  const password = document.getElementById('aeries-password').value;
  const rememberMe = document.getElementById('remember-me').checked;
  const errorEl = document.getElementById('login-error');
  const loadingEl = document.getElementById('login-loading');
  const loginBtn = document.getElementById('login-button');

  if (!username || !password) {
    errorEl.textContent = 'Please enter both username and password.';
    errorEl.classList.remove('hidden');
    return;
  }

  errorEl.classList.add('hidden');
  loadingEl.classList.remove('hidden');
  loginBtn.disabled = true;

  try {
    const functions = getFunctions();
    const authenticateTeacher = httpsCallable(functions, 'authenticateTeacher');
    const result = await authenticateTeacher({ username, password });
    const { token, uid, displayName, validated } = result.data;

    // Sign in to Firebase with custom token
    await signInWithCustomToken(auth, token);

    teacherUid = uid;
    teacherName = displayName;

    if (!validated) {
      // Credentials couldn't be validated against Aeries (server unreachable)
      // Still proceed — validation will happen when sync runs
      console.warn('[login] Aeries validation skipped — credentials stored for later verification');
    }

    // Store binding if "Remember me" is checked
    if (rememberMe) {
      localStorage.setItem('kioskTeacherUid', uid);
      localStorage.setItem('kioskTeacherName', displayName);
      localStorage.setItem('kioskAuthToken', token);
    }

    // Check if PIN already set for this teacher on this device
    const existingPin = localStorage.getItem('kioskTeacherPin');
    if (existingPin) {
      teacherPin = existingPin;
      showKioskMode();
    } else {
      showPinSetup();
    }
  } catch (err) {
    console.error('[login] Error:', err);
    const msg = err.message || 'Login failed. Please check your credentials.';
    errorEl.textContent = msg;
    errorEl.classList.remove('hidden');
  } finally {
    loadingEl.classList.add('hidden');
    loginBtn.disabled = false;
  }
}
```
